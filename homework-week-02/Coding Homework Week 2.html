Note on Species variable

The instructions of this assignment say to use the categorical variable Species, but the instructions also saythat we should have six control variables.

Normally, I would convert each value of Species into a separate dummy variable, which would result in 11 control variables (omitting one category to avoid collinearity). But to align with the instructions, I am going to encode the values of species to integers based on alphabetical order.

Data Processing

We apply the following steps to prepare the data:

  • convert Species to a numerical variable
  • apply a min-max function to restrict all variables to [0, 1]
  • separate the data into X and Y data sets
  • split into training and testing sets using an 80% random sample of the data for training

# define some helper functions
def initialize_params(seed=1234):
    random.seed(seed)
    w = [random.uniform(-1,1) for _ in range(X_train.shape[1])]
    b = random.uniform(-1,1)
    a = random.uniform(0,.1) # learning rate
    n = 50000 # number of gradient descent steps
    m = X_train.shape[0] # number of rows
    J = X_train.shape[1] # number of columns
    cost_history = [] # loss function history
    return w,b,a,n,m,J,cost_history

def predict(x_i,w,b):
    z = [x_i[i]*w[i] for i in range(len(x_i))] # dot product using lists
    return(sum(z)+b)

def predict_vec(x,w,b):
    return(x@w + b) # matrix multiplication

# calculating gradient descent with for loops algorithm
w,b,a,n,m,J,cost_history = initialize_params()
y = Y_train.to_list()

t0 = time.time()
for k in range(n):
	y_hat,resid,resid2=[],[],[]
	residx=[[] for _ in range(J)] # Create J separate empty lists
	
	for i in range(m):
		x_i = X_train.iloc[i,:].to_list()

		y_hat.append(predict(x_i,w,b)) # make predictions with existing params
		resid.append(y_hat[i] - y[i]) # calculate residuals
		resid2.append(resid[i]**2) # residuals squared
		
		for j in range(J):
			residx[j].append(resid[i] * x_i[j]) # calculate residual times x_j
			
	b = b - (a/m * sum(resid[:])) # gradient descent for constant term
	
	for j in range(J):
		w[j] = w[j] - (a/m * sum(residx[j])) # gradient descent for each control var
	
	cost = 1/(2*m) * sum(resid2) # calculate total cost
	if k%10000==0: print(f'Iteration {k}: Total Cost: {cost:0.6f}.')

	cost_history.append(cost)

time_for_loop_algo = time.time() - t0 # Time to run
cost_history_for_loop = cost_history # Save cost history for comparison
print(f'Final cost (for loop algo): {cost_history[-1]:0.6f}.')

# calculating gradient descent with vectorized algorithm
w,b,a,n,m,J,cost_history = initialize_params()
w = np.array(w) # convert w to numpy array
b = np.array(b) # convert b to numpy array

t0 = time.time()
for k in range(n):
    y_hat = predict_vec(X_train,w,b) # make predictions with existing params
    b = b - (a/m * np.sum(y_hat - Y_train))
    w = w - (a/m * X_train.T @ (y_hat - Y_train))
    cost = 1/(2*m) * (y_hat - Y_train).T @ (y_hat - Y_train) # calculate total cost
    if k%10000==0: print(f'Iteration {k}: Total Cost: {cost:0.6f}.')

    cost_history.append(cost)

time_vec_algo = time.time() - t0 # Time to run
cost_history_vec = cost_history # Save cost history for comparison
print(f'Final cost (vectorized algo): {cost_history[-1]:0.6f}.')
